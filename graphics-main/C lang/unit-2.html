                            scanf("%d",&a[i]);
                        }
                        for(i=0;i < size;i++){
                            for(j=i+1;j < size;j++){
                                if(a[i]>a[j]){
                                    swap = a[i];
                                    a[i] = a[j];

                            }

                        }

                    }

Matrix Multiplication 

            
                #include <stdio.h>
                    int main(){
                    int m, i, j, k;
                    int mat1[10][10], mat2[10][10], res[10][10];
                    scanf("%d", &m);
                    for (i = 0; i < m; i++){
                        for (j = 0; j < m; j++){
                            scanf("%d", &mat1[i][j]);
                        }
                    }
                    for (i = 0; i < m; i++){
                        for (j = 0; j < m; j++){
                            scanf("%d", &mat2[i][j]);
                        }
                    }
                    // multiply the two matrices
                    for (i = 0; i < m; i++){
                        for (j = 0; j < m; j++){
                            res[i][j] = 0;
                            for (k = 0; k < m; k++){
                                res[i][j] += mat1[i][k] * mat2[k][j];
                            }
                        }
                    }
                    for (i = 0; i < m; i++){
                        for (j = 0; j < m; j++){
                            printf("%d ", res[i][j]);
                        }
                        printf("\n");
                    }
                    return 0;
                    }

Recursion 2 : Fibonacci Series

                    int fibo(int a){
                        if(a == 1){
                            return 0;
                        }
                        else if( a == 2){
                            return 1;
                        }
                        else if(a!=2){
                            sum = a1+a2;
                            a1 = a2;
                            a2 = sum;
                            a--;
                            fibo(a);
                        }
                        return sum;
                    }

                    }
                    int computePower(int a, int n) {
                        if (n == 0) {
                            return 1;
                        }
                        if (n == 1) {
                            return a;
                        }
                        return a * computePower(a, n - 1);
                    }
