                  printf("Enter an Integer:\n");
                  scanf("%d",&a);
                  while ((a & 1) !=1) {
                      c++;
                      a = a>>1;
                  }
                  (c == 0)?printf("There are no Trailing Zeros."):printf("There are %d Trailing Zeros.",c);
                  return 0;
              }

Decimal to Binary Conversion

          
            #include 
              int main(){
                  int n, m=1, rem, bin=0,k;
                  printf("Enter the decimal number\n");
                  scanf("%d", &n);
                  k = n;
                  while(n > 0){
                      rem = n % 2;
                      bin = bin + (rem * m);
                      m = m * 10;
                      n = n / 2;
                  }
                  printf("The binary equivalent of decimal number %d is %d\n", k, bin);
                  return 0;
              }
